**Date**: April 21, 2026  
**Objective**: Implement PCI-compliant ISO 8583 persistence inside jPOS-EE  
**Target**: Replace webhook approach with native jPOS module  
**Status**: Planning & Phased Implementation

---

## ⚠️ Critical Considerations Before Starting

### Why This Matters for PCI Compliance

```
PCI-DSS Requirement 10.2 & 10.3:
✅ Log and monitor all access to cardholder data
✅ Protect log files from deletion or modification
✅ Promptly review and act on exceptions

Current Webhook Approach (Non-Compliant):
❌ Original ISO message NOT stored
❌ Can't prove what was sent/received
❌ Missing fields for audit trail
❌ Not enterprise-grade logging

jPOS-EE Native Approach (Compliant):
✅ Raw request/response stored (CLOB)
✅ Complete audit trail
✅ Transaction replay capability
✅ Enterprise-grade persistence
✅ Meets PCI-DSS 10.x requirements
```

```sql
CREATE TABLE ISO_TRANSACTIONS (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    
    -- ISO Core Fields (Parsed)
    MTI VARCHAR2(4) NOT NULL,           -- Message Type Indicator
    PAN VARCHAR2(19),                  -- Primary Account Number (masked)
    PAN_MASKED VARCHAR2(19),           -- PAN masked: XXXXXX****1234
    PROCESSING_CODE VARCHAR2(6),       -- Transaction type
    AMOUNT NUMBER(15,2) NOT NULL,      -- Transaction amount
    CURRENCY_CODE VARCHAR2(3),         -- ISO 4217 code
    STAN VARCHAR2(12) NOT NULL UNIQUE, -- System Trace Audit Number
    RRN VARCHAR2(12),                  -- Retrieval Reference Number
    
    -- Response Data
    RESPONSE_CODE VARCHAR2(2),         -- Field 39: Response code
    AUTH_ID_RESP VARCHAR2(6),          -- Authorization ID Response
    
    -- Raw Messages (Full Audit Trail)
    RAW_REQUEST CLOB NOT NULL,         -- Complete ISO request (packed)
    RAW_RESPONSE CLOB,                 -- Complete ISO response (packed)
    
    -- Merchant & Terminal Info
    MERCHANT_ID VARCHAR2(15),
    TERMINAL_ID VARCHAR2(8),
    
    -- Processing Status
    STATUS VARCHAR2(20),               -- RECEIVED, PROCESSED, FAILED, DECLINED
    ERROR_MESSAGE VARCHAR2(500),       -- If failed, error details
    
    -- Timestamps & Audit
    CREATED_AT TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    UPDATED_AT TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PROCESSED_AT TIMESTAMP,
    CREATED_BY VARCHAR2(50),           -- Who created record
    UPDATED_BY VARCHAR2(50),           -- Who last updated
    
    -- Additional Audit Fields
    IP_ADDRESS VARCHAR2(45),           -- Source IP (IPv4 or IPv6)
    SESSION_ID VARCHAR2(50),           -- jPOS session ID
    
    -- PCI Compliance Fields
    SENSITIVE_DATA_ENCRYPTED NUMBER DEFAULT 1, -- 1=encrypted, 0=test only
    COMPLIANCE_CHECKED NUMBER DEFAULT 0,       -- 1=verified PCI compliant
    
    CONSTRAINT chk_status CHECK (STATUS IN ('RECEIVED', 'PROCESSED', 'FAILED', 'DECLINED'))
);

-- Indexes for performance & compliance queries
CREATE INDEX IDX_STAN ON ISO_TRANSACTIONS(STAN);
CREATE INDEX IDX_RRN ON ISO_TRANSACTIONS(RRN);
CREATE INDEX IDX_PAN_MASKED ON ISO_TRANSACTIONS(PAN_MASKED);
CREATE INDEX IDX_MERCHANT ON ISO_TRANSACTIONS(MERCHANT_ID);
CREATE INDEX IDX_CREATED_AT ON ISO_TRANSACTIONS(CREATED_AT);
CREATE INDEX IDX_STATUS ON ISO_TRANSACTIONS(STATUS);

-- Audit table for compliance (tracks modifications)
CREATE TABLE ISO_TRANSACTIONS_AUDIT (
    AUDIT_ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    TXN_ID NUMBER NOT NULL,
    FIELD_NAME VARCHAR2(50),
    OLD_VALUE VARCHAR2(500),
    NEW_VALUE VARCHAR2(500),
    CHANGED_BY VARCHAR2(50),
    CHANGED_AT TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
```

#### 3.1 PersistRequest Participant
**File**: `jpos-ee/src/main/java/org/cms/jposee/participant/PersistRequest.java`

```java
package org.cms.jposee.participant;

import org.jpos.core.Configurable;
import org.jpos.core.Configuration;
import org.jpos.core.ConfigurationException;
import org.jpos.transaction.Context;
import org.jpos.transaction.TransactionParticipant;
import org.cms.jposee.entity.IsoTransaction;
import org.cms.jposee.repository.IsoTransactionRepository;
import org.cms.jposee.util.IsoUtil;
import org.jpos.iso.ISOMsg;
import org.jpos.core.XmlConfigurator;

import javax.persistence.EntityManager;
import javax.persistence.EntityManagerFactory;
import javax.persistence.Persistence;
import java.math.BigDecimal;

public class PersistRequest implements TransactionParticipant, Configurable {
    
    private EntityManagerFactory emf;
    private Configuration cfg;
    
    @Override
    public void setConfiguration(Configuration cfg) throws ConfigurationException {
        this.cfg = cfg;
        // Initialize EntityManager from persistence.xml
        this.emf = Persistence.createEntityManagerFactory("jposee-persistence");
    }
    
    @Override
    public int prepare(long id, Serializable context) {
        Context ctx = (Context) context;
        
        try {
            // Get the incoming ISO message
            ISOMsg request = (ISOMsg) ctx.get("REQUEST");
            if (request == null) {
                return ABORTED;
            }
            
            // Create entity for persistence
            IsoTransaction txn = new IsoTransaction();
            
            // Extract ISO Fields (safely)
            txn.setMti(IsoUtil.get(request, 0));  // MTI
            txn.setPan(maskPan(IsoUtil.get(request, 2)));  // PAN (masked)
            txn.setPanMasked(maskPan(IsoUtil.get(request, 2)));
            txn.setProcessingCode(IsoUtil.get(request, 3));
            txn.setAmount(new BigDecimal(IsoUtil.get(request, 4, "0")));
            txn.setCurrencyCode(IsoUtil.get(request, 49, "840"));  // ISO 4217
            txn.setStan(IsoUtil.get(request, 11));  // System Trace Audit Number
            txn.setMerchantId(IsoUtil.get(request, 42));
            txn.setTerminalId(IsoUtil.get(request, 41));
            
            // Store raw message (CRITICAL FOR PCI COMPLIANCE)
            try {
                txn.setRawRequest(request.pack());  // Full ISO message
            } catch (Exception e) {
                logger.error("Failed to pack request", e);
                return ABORTED;
            }
            
            // Set audit fields
            txn.setCreatedBy("jPOS_Channel");
            txn.setIpAddress((String) ctx.get("SOURCE_IP"));
            txn.setSessionId((String) ctx.get("SESSION_ID"));
            
            // Persist to database
            EntityManager em = emf.createEntityManager();
            try {
                em.getTransaction().begin();
                em.persist(txn);
                em.getTransaction().commit();
                
                // Store ID in context for later update
                ctx.put("TXN_ID", txn.getId());
                ctx.put("TXN_OBJECT", txn);
                
                return PREPARED;
                
            } finally {
                em.close();
            }
            
        } catch (Exception e) {
            logger.error("PersistRequest failed", e);
            return ABORTED;
        }
    }
    
    @Override
    public void commit(long id, Serializable context) {
        // Request has been successfully processed
        Context ctx = (Context) context;
        logger.info("Transaction " + ctx.get("TXN_ID") + " committed");
    }
    
    @Override
    public void abort(long id, Serializable context) {
        // Transaction failed - mark in database
        Context ctx = (Context) context;
        Long txnId = (Long) ctx.get("TXN_ID");
        
        if (txnId != null) {
            try {
                EntityManager em = emf.createEntityManager();
                IsoTransaction txn = em.find(IsoTransaction.class, txnId);
                if (txn != null) {
                    txn.setStatus("FAILED");
                    txn.setErrorMessage("Transaction aborted");
                    em.getTransaction().begin();
                    em.merge(txn);
                    em.getTransaction().commit();
                }
                em.close();
            } catch (Exception e) {
                logger.error("Failed to update transaction status", e);
            }
        }
    }
    
    private String maskPan(String pan) {
        if (pan == null || pan.length() < 8) {
            return pan;
        }
        // Show first 6 and last 4 digits
        return pan.substring(0, 6) + "****" + pan.substring(pan.length() - 4);
    }
}
```

**File**: `jpos-ee/src/main/java/org/cms/jposee/participant/UpdateResponse.java`

```java
package org.cms.jposee.participant;

import org.jpos.core.Configurable;
import org.jpos.core.Configuration;
import org.jpos.core.ConfigurationException;
import org.jpos.transaction.Context;
import org.jpos.transaction.TransactionParticipant;
import org.cms.jposee.entity.IsoTransaction;
import org.cms.jposee.util.IsoUtil;
import org.jpos.iso.ISOMsg;

import javax.persistence.EntityManager;
import javax.persistence.EntityManagerFactory;
import javax.persistence.Persistence;
import java.time.LocalDateTime;

public class UpdateResponse implements TransactionParticipant, Configurable {
    
    private EntityManagerFactory emf;
    
    @Override
    public void setConfiguration(Configuration cfg) throws ConfigurationException {
        this.emf = Persistence.createEntityManagerFactory("jposee-persistence");
    }
    
    @Override
    public int prepare(long id, Serializable context) {
        Context ctx = (Context) context;
        
        try {
            Long txnId = (Long) ctx.get("TXN_ID");
            ISOMsg response = (ISOMsg) ctx.get("RESPONSE");
            
            if (txnId == null || response == null) {
                return PREPARED;
            }
            
            EntityManager em = emf.createEntityManager();
            try {
                // Load transaction from database
                IsoTransaction txn = em.find(IsoTransaction.class, txnId);
                
                if (txn != null) {
                    // Update with response data
                    txn.setResponseCode(IsoUtil.get(response, 39));  // Response Code
                    txn.setAuthIdResp(IsoUtil.get(response, 38));    // Auth ID Response
                    
                    // Store raw response
                    try {
                        txn.setRawResponse(response.pack());  // Full ISO response
                    } catch (Exception e) {
                        logger.error("Failed to pack response", e);
                    }
                    
                    // Update status based on response code
                    String responseCode = IsoUtil.get(response, 39, "99");
                    if ("00".equals(responseCode)) {
                        txn.setStatus("PROCESSED");
                    } else {
                        txn.setStatus("DECLINED");
                        txn.setErrorMessage("Response code: " + responseCode);
                    }
                    
                    txn.setProcessedAt(LocalDateTime.now());
                    txn.setUpdatedBy("jPOS_Channel");
                    txn.setComplianceChecked(1);  // Mark as compliance verified
                    
                    // Merge changes
                    em.getTransaction().begin();
                    em.merge(txn);
                    em.getTransaction().commit();
                    
                    logger.info("Transaction " + txnId + " updated with response: " + responseCode);
                }
                
                return PREPARED;
                
            } finally {
                em.close();
            }
            
        } catch (Exception e) {
            logger.error("UpdateResponse failed", e);
            return PREPARED;  // Don't abort - response already sent
        }
    }
    
    @Override
    public void commit(long id, Serializable context) {
        // Update has been committed
    }
    
    @Override
    public void abort(long id, Serializable context) {
        // Even if abort, response was already updated
    }
}
```

#### 3.3 IsoUtil Helper
**File**: `jpos-ee/src/main/java/org/cms/jposee/util/IsoUtil.java`

```java
package org.cms.jposee.util;

import org.jpos.iso.ISOMsg;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class IsoUtil {
    
    private static Logger logger = LoggerFactory.getLogger(IsoUtil.class);
    
    /**
     * Safely extract ISO field - returns null if not present
     */
    public static String get(ISOMsg msg, int fieldNum) {
        try {
            if (msg.hasField(fieldNum)) {
                return msg.getString(fieldNum);
            }
        } catch (Exception e) {
            logger.debug("Error getting field " + fieldNum, e);
        }
        return null;
    }
    
    /**
     * Safely extract ISO field with default value
     */
    public static String get(ISOMsg msg, int fieldNum, String defaultValue) {
        String value = get(msg, fieldNum);
        return value != null ? value : defaultValue;
    }
    
    /**
     * Map MTI code to transaction type
     */
    public static String getMtiType(String mti) {
        if (mti == null) return "UNKNOWN";
        switch (mti) {
            case "0100": return "AUTH";
            case "0110": return "AUTH_RESPONSE";
            case "0200": return "FINANCIAL";
            case "0210": return "FINANCIAL_RESPONSE";
            case "0400": return "REVERSAL";
            case "0410": return "REVERSAL_RESPONSE";
            default: return "OTHER";
        }
    }
}
```
